017package net.ljcomputing.sr.fx.treeview;
018
019import net.ljcomputing.model.Model;
020import net.ljcomputing.sr.fx.menu.item.AddElapsedTaskMenuItem;
021import net.ljcomputing.sr.fx.menu.item.AddNewWbsMenuItem;
022import net.ljcomputing.sr.fx.menu.item.AddTaskMenuItem;
023import net.ljcomputing.sr.fx.menu.item.AddWbsMenuItem;
024import net.ljcomputing.sr.fx.menu.item.DeleteWbsMenuItem;
025import net.ljcomputing.sr.fx.menu.item.EditWbsMenuItem;
026import net.ljcomputing.sr.model.Activity;
027import net.ljcomputing.sr.model.WorkBreakdownStructure;
028
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032import javafx.scene.control.ContextMenu;
033import javafx.scene.control.TreeCell;
034
035/**
036 * @author James G. Willmore
037 *
038 */
039public class WbsTreeCell extends TreeCell<Model> {
040  @SuppressWarnings("unused")
041  private final Logger logger = LoggerFactory.getLogger(WbsTreeCell.class);
042  
043  public WbsTreeCell(WbsTreeView treeView) {
044    super();
045
046    selectedProperty().addListener(listener -> {
047      ContextMenu contextMenu = new ContextMenu();
048      
049      if (treeView.getRoot().getValue().equals(getItem())) {
050        contextMenu.getItems().addAll(new AddNewWbsMenuItem(treeView));
051      }
052      
053      if (getItem() instanceof WorkBreakdownStructure && !treeView.getRoot().getValue().equals(getItem())) {
054        contextMenu.getItems().addAll(new AddWbsMenuItem(getItem(), treeView));
055        contextMenu.getItems().addAll(new DeleteWbsMenuItem(getItem(), treeView));
056      }
057
058      if (getItem() instanceof Activity) {
059        contextMenu.getItems().addAll(new AddTaskMenuItem(getItem(), treeView, treeView.getTaskTableController()));
060        contextMenu.getItems().addAll(new AddElapsedTaskMenuItem(getItem(), treeView, treeView.getTaskTableController()));
061        contextMenu.getItems().addAll(new EditWbsMenuItem(getItem(), treeView));
062        contextMenu.getItems().addAll(new DeleteWbsMenuItem(getItem(), treeView));
063      }
064      
065      setContextMenu(contextMenu);
066    });
067  }
068
069  @Override
070  public void updateItem(Model item, boolean empty) {
071    super.updateItem(item, empty);
072    if(!empty) {
073      setText(getItemText(item));
074    } else {
075      //set graphic to null for empty cells - rendering gets fouled if it is not set
076      setGraphic(null);
077      //set text to null for empty cells - rendering gets fouled if it is not set
078      setText(null);
079    }
080  }
081  
082  /**
083   * Gets the item text.
084   *
085   * @param item the item
086   * @return the item text
087   */
088  private String getItemText(Object item) {
089    switch(item.getClass().getSimpleName()) {
090      case "WorkBreakdownStructure":
091        return ((WorkBreakdownStructure)item).getName();
092      case "Activity":
093        return ((Activity)item).getName();
094      default:
095        return "";
096    }
097  }
098}
