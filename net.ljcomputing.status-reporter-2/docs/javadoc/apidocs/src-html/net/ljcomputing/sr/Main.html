044/**
045 * Status Reporter main application class.
046 * 
047 * @author James G. Willmore
048 *
049 */
050public class Main extends Application {
051
052  /** The logger. */
053  private final Logger logger = LoggerFactory.getLogger(Main.class);
054
055  /**
056   * @see javafx.application.Application#start(javafx.stage.Stage)
057   */
058  @Override
059  public void start(Stage primaryStage) throws IOException {
060    logger.debug("starting application");
061    Thread.setDefaultUncaughtExceptionHandler((t, e) -> Platform.runLater(() -> showErrorDialog(t, e)));
062    Thread.currentThread().setUncaughtExceptionHandler(this::showErrorDialog);
063    
064    setUserAgentStylesheet(STYLESHEET_MODENA);
065    Parent root = (Parent) FXMLLoader.load(getClass().getResource(StatusReporterConfig.MAIN_FXML));
066    Scene scene = new Scene(root);
067    primaryStage.setScene(scene);
068    primaryStage.setTitle(StatusReporterConfig.APP_NAME);
069    primaryStage.setOnCloseRequest(e -> {
070      logger.debug("exiting application");
071      Platform.exit();
072      System.exit(0);
073    });
074
075    setScreen(primaryStage);
076
077    primaryStage.setOnShown(e -> {
078      if(!PersistenceValidator.initialize()) {
079        Alert alert = new Alert(AlertType.ERROR);
080        alert.setContentText("An error occured initializing the database\n (is another instance running?)");
081        alert.showAndWait();
082        Platform.exit();
083        System.exit(1);
084      }
085    });
086
087    primaryStage.show();
088
089    logger.debug("primary stage showing");
090  }
091
092  // TODO - refactor
093  private void setScreen(Stage stage) {
094    ObservableList<Screen> screens = Screen.getScreens();
095    Screen screen = null;
096
097    for (Screen s : screens) {
098      screen = s;
099    }
100
101    Rectangle2D sbounds = screen.getVisualBounds();
102
103    double sw = sbounds.getMaxY();
104    double sh = sbounds.getMaxX();
105
106    listenToSizeInitialization(stage.widthProperty(), w -> stage.setX((sw - w) * 0.25));
107    listenToSizeInitialization(stage.heightProperty(), h -> stage.setY((sh - h) * 0.75));
108  }
109
110  private void listenToSizeInitialization(ObservableDoubleValue size, DoubleConsumer handler) {
111
112    ChangeListener<Number> listener = new ChangeListener<Number>() {
113      @Override
114      public void changed(ObservableValue<? extends Number> obs, Number oldSize, Number newSize) {
115        logger.debug("{}", newSize);
116        if (newSize.doubleValue() != Double.NaN) {
117          handler.accept(newSize.doubleValue());
118          size.removeListener(this);
119        }
120      }
121    };
122
123    size.addListener(listener);
124  }
125
126  /**
127   * The main method.
128   *
129   * @param args the arguments
130   */
131  public static void main(String... args) {
132    Application.launch(Main.class, args);
133  }
134
135  /**
136   * Show error dialog.
137   *
138   * @param thread the thread @param exception the exception
139   */
140  private void showErrorDialog(Thread thread, Throwable exception) {
141    new ErrorAlert().show((Exception) exception);
142  }
143}
