017package net.ljcomputing.sr.fx.dialog;
018
019import net.ljcomputing.sr.model.Activity;
020import net.ljcomputing.sr.model.WorkBreakdownStructure;
021import net.ljcomputing.sr.service.WorkBreakdownStructureService;
022import net.ljcomputing.sr.service.impl.ActivityServiceImpl;
023import net.ljcomputing.sr.service.impl.WorkBreakdownStructureServiceImpl;
024
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028import javafx.collections.FXCollections;
029import javafx.collections.ObservableList;
030import javafx.scene.control.ComboBox;
031import javafx.scene.control.Label;
032import javafx.scene.control.TextField;
033import javafx.util.StringConverter;
034
035/**
036 * A JavaFX activity data entry dialog.
037 * 
038 * @author James G. Willmore
039 *
040 */
041public class ActivityDataDialog
042    extends AbstractModelDataDialog<Activity, ActivityServiceImpl> {
043  
044  /** The logger. */
045  @SuppressWarnings("unused")
046  private final Logger logger = LoggerFactory.getLogger(ActivityDataDialog.class);
047  
048  /** The wbs service. */
049  private WorkBreakdownStructureService wbsService = new WorkBreakdownStructureServiceImpl();
050  
051  /** The wbs list. */
052  private ObservableList<WorkBreakdownStructure> wbsList;
053  
054  /** The lbl wbs. */
055  private Label lblWbs = new Label("WBS: ");
056  
057  /** The lbl name. */
058  private Label lblName = new Label("Name: ");
059  
060  /** The lbl description. */
061  private Label lblDescription = new Label("Description: ");
062  
063  /** The wbs. */
064  private ComboBox<WorkBreakdownStructure> wbsField;
065  
066  /** The name field. */
067  private TextField nameField;
068  
069  /** The description field. */
070  private TextField descriptionField;
071
072  /**
073   * Instantiates a new activity data dialog.
074   *
075   * @param item the item
076   * @throws Exception the exception
077   */
078  public ActivityDataDialog(Activity item) throws Exception {
079    super.item = item;
080
081    nameField = new TextField(item.getName());
082    descriptionField = new TextField(item.getDescription());
083    wbsField = new ComboBox<WorkBreakdownStructure>();
084
085    initWbsCombo();
086    wbsField.setValue(getSelectedWbs(item.getWbs()));
087
088    addContent(lblWbs, wbsField, lblName, nameField, lblDescription, descriptionField);
089  }
090
091  /**
092   * Update item.
093   *
094   * @return the activity
095   * @see net.ljcomputing.sr.fx.dialog.AbstractModelDataDialog#updateItem()
096   */
097  @Override
098  protected Activity updateItem() {
099    item = new Activity(nameField.getText(), wbsField.getValue().getId());
100    item.setDescription(descriptionField.getText());
101    
102    if(!"".equals(idField.getText())) {
103      item.setId(Integer.valueOf(idField.getText()));
104    }
105    
106    return item;
107  }
108  
109  /**
110   * Gets the selected wbs.
111   *
112   * @param selected the selected
113   * @return the selected wbs
114   */
115  private WorkBreakdownStructure getSelectedWbs(Integer selected) {
116    for(WorkBreakdownStructure item : wbsList) {
117      if(selected.equals(item.getId())) {
118        return item;
119      }
120    }
121    
122    return null;
123  }
124
125  /**
126   * Inits the wbs combo.
127   *
128   * @throws Exception the exception
129   */
130  private void initWbsCombo() throws Exception {
131    wbsField.setConverter(new WbsConverter());
132    populateWbsCombo();
133  }
134
135  /**
136   * Populate wbs combo.
137   *
138   * @throws Exception the exception
139   */
140  private void populateWbsCombo() throws Exception {
141    wbsList = FXCollections.observableList(wbsService.readAll());
142    wbsField.setItems(wbsList);
143  }
144  
145  /**
146   * Work breakdown structure String Converter.
147   */
148  class WbsConverter extends StringConverter<WorkBreakdownStructure> {
149
150    /**
151     * @see javafx.util.StringConverter#toString(java.lang.Object)
152     */
153    @Override
154    public String toString(WorkBreakdownStructure object) {
155      return object.getName();
156    }
157
158    /**
159     * @see javafx.util.StringConverter#fromString(java.lang.String)
160     */
161    @Override
162    public WorkBreakdownStructure fromString(String string) {
163      for(WorkBreakdownStructure item : wbsList) {
164        if(item.getName().equals(string)) {
165          return item;
166        }
167      }
168      
169      return null;
170    }
171  }
172}
