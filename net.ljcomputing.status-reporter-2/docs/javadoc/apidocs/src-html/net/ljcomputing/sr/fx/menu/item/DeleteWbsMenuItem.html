018package net.ljcomputing.sr.fx.menu.item;
019
020import net.ljcomputing.fx.alert.ErrorAlert;
021import net.ljcomputing.model.Model;
022import net.ljcomputing.sr.fx.controller.DeleteModelController;
023import net.ljcomputing.sr.fx.treeview.WbsTreeView;
024
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028import javafx.beans.binding.Bindings;
029import javafx.scene.control.TreeItem;
030
031/**
032 * The menu item to delete a new work breakdown structure.
033 * 
034 * @author James G. Willmore
035 *
036 */
037public class DeleteWbsMenuItem extends AbstractWbsMenuItem {
038  
039  /** The logger. */
040  @SuppressWarnings("unused")
041  private static final Logger logger = LoggerFactory.getLogger(DeleteWbsMenuItem.class);
042  
043  /** The item. */
044  private Model item;
045  
046  /** The tree view. */
047  private WbsTreeView treeView;
048  
049  /**
050   * Instantiates a new adds the activity menu item.
051   *
052   * @param item the item
053   * @param treeView the tree view
054   */
055  public DeleteWbsMenuItem(Model item, WbsTreeView treeView) {
056    super();
057    this.treeView = treeView;
058    this.item = item;
059    
060    textProperty().bind(Bindings.format("Delete %s", getItemText(item)));
061    setOnAction(event -> {
062      try {
063        deleteItem();
064      } catch (Exception e) {
065        new ErrorAlert().show(e);
066      }
067    });
068  }
069
070  private void deleteItem() {
071    try {
072      DeleteModelController controller = new DeleteModelController();
073      controller.delete(item);
074      
075      TreeItem<Model> ti = treeView.getSelectionModel().getSelectedItem();
076      ti.getParent().getChildren().remove(ti);
077    } catch (Exception e) {
078      new ErrorAlert().show(e);
079    }
080  }
081}
